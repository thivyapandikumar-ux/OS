#include&lt;stdio.h&gt;
#include&lt;sys/shm.h&gt;
#include&lt;sys/ipc.h&gt;
int main()
{
int child,shmid,i;
char *shmptr;
child=fork();
if(!child)
{
shmid=shmget(2041,32,0666|IPC_CREAT);
shmptr=shmat(shmid,0,0);
printf(&quot;\n PARENT PROCESS WRITING \n&quot;);
for(i=0;i&lt;10;i++)
{
shmptr[i]=&#39;a&#39;+i;
putchar(shmptr[i]);
}
printf(&quot;\n\n %s&quot;,shmptr);
wait(NULL);
}
else
{
shmid=shmget(2041,32,0666);
shmptr=shmat(shmid,0,0);
printf(&quot;\n CHILD PROCESS READING\n&quot;);
for(i=0;i&lt;10;i++)
putchar(shmptr[i]);
shmdt(NULL);
shmctl(shmid,IPC_RMID,NULL);
}
return 0
